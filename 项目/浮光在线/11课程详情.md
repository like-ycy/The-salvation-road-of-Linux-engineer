路由显示，`router/index.js`，代码：

```javascript
import {createRouter, createWebHistory} from 'vue-router'

// 路由列表
const routes = [
  {
    meta:{
        title: "首页-浮光在线",
        keepAlive: true
    },
    path: '/',         // uri访问地址
    name: "Home",
    component: ()=> import("../views/Home.vue")
  },
  {
    meta:{
        title: "用户登录-浮光在线",
        keepAlive: true
    },
    path:'/login',      // uri访问地址
    name: "Login",
    component: ()=> import("../views/Login.vue")
  }
  ,{
      meta:{
        title: "用户注册-浮光在线",
        keepAlive: true
      },
      path: '/register',
      name: "Register",            // 路由名称
      component: ()=> import("../views/Register.vue"),         // uri绑定的组件页面
    },{
      meta:{
        title: "项目课-浮光在线",
        keepAlive: true
      },
      path: '/project',
      name: "Course",            // 路由名称
      component: ()=> import("../views/Course.vue"),         // uri绑定的组件页面
    },{
      meta:{
        title: "项目课",
        keepAlive: true
      },
      path: '/project/:id',     // :id vue的路径参数，代表了课程的ID
      name: "Info",
      component: ()=> import("../views/Info.vue"),
    }
]

// 路由对象实例化
const router = createRouter({
  // history, 指定路由的模式
  history: createWebHistory(),
  // 路由列表
  routes,
});


// 暴露路由对象
export default router
```

### 视频播放器[bug]

针对客户端的课程详情页的左上角内容，我们可以显示课程的详情图片，如果课程的介绍视频，也可以优先显示视频，

当然播放视频，肯定需要对应的播放器。市面上很多：百度云、腾讯云、网易云、阿里云、又拍云、七牛云或者其他第三方。

阿里云播放器，官网：https://player.alicdn.com/aliplayer/index.html

当然，接下来使用的播放器组件，选择使用了vue-video视频播放组件，所以我们需要先预安装。

vue-video-player地址：https://github.com/LarchLiu/vue3-video-player

npm包列表：

演示效果：https://cloudgeektech.com/vue3-video-player/

安装依赖

```bash
cd fuguangweb/
yarn add @cloudgeek/vue3-video-player --save
```

在main.js中注册全局加载播放器组件

```javascript
import { createApp } from 'vue'
import App from './App.vue'
import router from "./router/index.js"
import store from "./store/index.js"
import ElementPlus from 'element-plus'
import 'element-plus/lib/theme-chalk/index.css';
import Vue3VideoPlayer from '@cloudgeek/vue3-video-player'
import '@cloudgeek/vue3-video-player/dist/vue3-video-player.css'

createApp(App).use(router).use(store).use(ElementPlus).use(Vue3VideoPlayer, {
  lang: 'zh-CN'
}).mount('#app')

```



Info.vue页面组件中调用vue3-video-player播放器，代码：

```vue
          <div class="wrap-left">
            <vue3-video-player @play="onPlay($event)" @pause="onPause($event)" cover="/src/assets/course-1.png" src="/src/assets/1.mp4"></vue3-video-player>
          </div>
```

```vue
<script setup>
import {reactive} from "vue"
import {useRoute} from "vue-router"
import Header from "../components/Header.vue"
import Footer from "../components/Footer.vue"

let route = useRoute()

const state = reactive({
  course_id: route.params.id,
  tabIndex: 2,
})


// 播放器开始播放视频的回调函数
let onPlay = (event)=>{
    console.log("播放")
    console.log(event);
    console.log(event.target.currentTime); // 当前播放的视频时间进度，可以用于跟踪播放视频的视频
    console.log(event.target.duration); // 当前视频的总时长，可以用于跟踪播放视频的视频
}

// 播放器暂停播放视频的回调函数
let onPause = (event)=>{
    console.log("暂停")
    console.log(event.target.currentTime); // 当前播放的视频时间进度，可以用于跟踪播放视频的视频
    console.log(event.target.duration); // 当前视频的总时长，可以用于跟踪播放视频的视频
}

</script>
```

如果安装和使用上面插件，没有效果，浏览器console终端出现以下错误：

```bash
TypeError: videoEl.canPlayType is not a function
```

解决方法如下，`把自己客户端项目根目录下的package.json的改名或者备份删除，从课件素材里面找到package.json和pageage-lock.json和yarn.lock复制到客户端项目根目录下，终端下直接在客户端根目录下yarn重新安装node_modules即可。当然可以提前删除node_module，防止出现缓存，导致安装不生效。`
